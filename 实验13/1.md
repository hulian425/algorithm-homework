```c++
#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

const int MAX_N = 1e7;
int a[MAX_N];
int main()
{
    int n;
    cin >> n;
    for (int i = 1; i <= n; i++)
    {
        int t;
        cin >> t;
        a[t]++;
    }
    int j = 0;
    for (int i = 0; i < MAX_N; i++)
    {
        if (a[i] > a[j])
            j = i;
    }

    printf("%d\n", j);
}
```
